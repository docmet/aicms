```python
Revision ID: 20260307_0600
Revises: 20260307_0500
Create Date: 2026-03-07 06:00:00.000000
"""

from alembic import op
from sqlalchemy import text

revision = "20260307_0600"
down_revision = "20260307_0500"
branch_labels = None
depends_on = None


def upgrade() -> None:
    # Partial unique index — only enforces uniqueness for non-deleted pages.
    # This allows re-creation of a slug after soft-deletion (is_deleted=True rows
    # are excluded from the constraint by the WHERE clause).
    op.create_index(
        "uq_pages_site_slug_active",
        "pages",
        ["site_id", "slug"],
        unique=True,
        postgresql_where=text("is_deleted = false"),
    )


def downgrade() -> None:
    op.drop_index("uq_pages_site_slug_active", table_name="pages")
```

The current `create_page` endpoint has a SELECT-then-INSERT pattern (lines ~85-98). This remains as a fast-path check. Add an `IntegrityError` catch around the commit to handle race conditions.

Add `IntegrityError` to the sqlalchemy imports at the top of the file:
```python
from sqlalchemy.exc import IntegrityError
```

Wrap the commit and refresh in a try/except block:
```python
db_page = Page(**page_in.model_dump(), site_id=site_id)
db.add(db_page)
try:
    await db.commit()
    await db.refresh(db_page)
except IntegrityError:
    await db.rollback()
    raise HTTPException(
        status_code=status.HTTP_409_CONFLICT,
        detail="A page with this slug already exists in this site",
    )
return db_page
```

The existing SELECT check on lines ~85-92 can be left in place for fast-path 400 response on simple single-request duplicates. The IntegrityError catch above handles the race condition case.
